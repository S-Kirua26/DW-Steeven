// permet de calculer la factrielle d'un nombre

function factorielle($nombre)
{
    $facto = 1;

    for ($i = 2; $i <= $nombre; $i++)
    {
        $facto = $facto * $i; //$facto *=$i;
    }
    return $facto;
}
